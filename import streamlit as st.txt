def safe_parse(response):
    try:
        if isinstance(response, str): response = json.loads(response)
        if isinstance(response, dict):
            content = response.get("content", response)
            if isinstance(content, str):
                parsed = json.loads(content)
                if isinstance(parsed, str): parsed = json.loads(parsed)
                return parsed
            return content
    except: return {"error": "Parse Error"}
    return {}

def build_messages():
    msgs = []
    for m in st.session_state.messages:
        role = "analyst" if m["role"] == "assistant" else m["role"]
        msgs.append({"role": role, "content": [{"type": "text", "text": m["content"]}]})
    if msgs and msgs[0]["role"] == "analyst": msgs.pop(0)
    return msgs

def extract_top_n(text):
    match = re.search(r"トップ\s*(\d+)", text)
    return int(match.group(1)) if match else None

def detect_ambiguous_query(text):
    return any(k in text for k in ["と", "、", "及び", "それぞれ"])

def normalize_query(text):
    if "月" in text and "店舗" in text:
        return "店舗別・カテゴリ別の売上を月ごとに表示"
    return text
