#sendAllWithWebhook.gs
/*
n8n用のデータを加工し、請求書単位でn8nに送信
JSON形式で請求書情報を入力シート（現在の名称は複数明細）から
シートの構成の情報が書かれたコンフィグ
*/
function sendAllWithWebhook(){
  // --- 基本設定 ---

  const WEBHOOK_URL = 'https://app-n8n-nat-e2fmhrece2cbc7h8.japanwest-01.azurewebsites.net/webhook/receive-invoice-from-gas';
  
  const INVOICE_SHEET_NAME = '【山本追加】複数明細'; // シート名を変更
  const CONFIG_SHEET_NAME = '【山本追加】コンフィグ';
  
  // --- コンフィグシートのセル設定（画像に合わせて修正） ---
  const CFG_HEADER_ROW = 'D5'; // ヘッダー行番号
  const CFG_START_ROW  = 'D7'; // 値開始行番号

  const CFG_KEY_COL    = 'C9';  // 請求書キーの列 (例: B)
  
  const CFG_MAIN_START = 'C10'; // 請求書情報ヘッダー開始列 (例: C)
  const CFG_MAIN_END   = 'C11'; // 請求書情報ヘッダー終了列 (例: N)
  
  const CFG_DTL_START  = 'C12'; // 明細情報ヘッダー開始列 (例: O)
  const CFG_DTL_END    = 'C13'; // 明細情報ヘッダー終了列 (例: U)
  
  const CFG_URL_COL    = 'C14'; // URLの列 (例: V)

 //山本追加分
  const CFG_VRTI_START = 'C15';
  const CFG_VRTI_END   = 'C16';
 //end

  // JSON出力時の固定キー名
  const KEY_NAME_DETAILS = '明細';
  const KEY_NAME_URLS    = 'URL';
  // ----------------

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetData = ss.getSheetByName(INVOICE_SHEET_NAME);
  const sheetConfig = ss.getSheetByName(CONFIG_SHEET_NAME);

  if (!sheetData || !sheetConfig) {
    console.error("シートが見つかりません。シート名を確認してください。");
    return;
  }

  // 1. コンフィグ情報の取得
  const headerRow = sheetConfig.getRange(CFG_HEADER_ROW).getValue();
  const startRow = sheetConfig.getRange(CFG_START_ROW).getValue();
  
  // 列定義の読み込み
  const keyColStr = sheetConfig.getRange(CFG_KEY_COL).getValue().toString();
  
  const mainStart = sheetConfig.getRange(CFG_MAIN_START).getValue().toString();
  const mainEnd   = sheetConfig.getRange(CFG_MAIN_END).getValue().toString();
  const mainRangeStr = `${mainStart}:${mainEnd}`; // "C:N" の形式を作成
  
  const dtlStart  = sheetConfig.getRange(CFG_DTL_START).getValue().toString();
  const dtlEnd    = sheetConfig.getRange(CFG_DTL_END).getValue().toString();
  const dtlRangeStr = `${dtlStart}:${dtlEnd}`; // "O:U" の形式を作成

  const urlColStr = sheetConfig.getRange(CFG_URL_COL).getValue().toString();

  // --- 山本追加分：コンフィグ取得 ---
  const vrtiStart = sheetConfig.getRange(CFG_VRTI_START).getValue().toString();
  const vrtiEnd   = sheetConfig.getRange(CFG_VRTI_END).getValue().toString();
  const vrtiRangeStr = `${vrtiStart}:${vrtiEnd}`; 
  // ------------------------------

  // バリデーション
  const lastRow = sheetData.getLastRow();
  const lastCol = sheetData.getLastColumn();
  if (typeof startRow !== 'number' || startRow > lastRow || startRow < 1) {
    console.log("送信対象のデータがありません。");
    return;
  }

  // 2. 列番号の解析（A列→0, B列→1 に変換）
  // ヘルパー関数: 列文字列("A"や"A:C")をインデックス配列([0]や[0,1,2])に変換
  const parseColString = (str) => {
    const parts = str.split(':');
    const colToIdx = (char) => {
      let idx = 0;
      // 列名が2文字以上(AAなど)の場合にも対応
      const cleanChar = char.trim().toUpperCase();
      for (let i = 0; i < cleanChar.length; i++) {
        idx = idx * 26 + cleanChar.charCodeAt(i) - 64;
      }
      return idx - 1; // 0-based index
    };
    
    if (parts.length === 1) {
      return [colToIdx(parts[0])];
    } else if (parts.length === 2) {
      const start = colToIdx(parts[0]);
      const end = colToIdx(parts[1]);
      const indices = [];
      for (let i = start; i <= end; i++) indices.push(i);
      return indices;
    }
    return [];
  };

  const keyIdx = parseColString(keyColStr)[0]; // 単一
  const mainIndices = parseColString(mainRangeStr); // 配列
  const dtlIndices = parseColString(dtlRangeStr);   // 配列
  const urlIdx = parseColString(urlColStr)[0]; // 単一

  // 3. データとヘッダーの取得
  const headers = sheetData.getRange(headerRow, 1, 1, lastCol).getValues()[0];
  const numRows = lastRow - startRow + 1;
  const values = sheetData.getRange(startRow, 1, numRows, lastCol).getValues();

  // --- 山本追加分：インデックス解析 ---
  const vrtiIndices = parseColString(vrtiRangeStr);
  // -------------------------------

  // 4. グルーピング処理
  const groupedData = {};

  for (let i = 0; i < values.length; i++) {
    const row = values[i];
    const currentRowNum = startRow + i;
    
    // キー取得
    const groupKey = row[keyIdx];
    
    // キーが空の場合はスキップ
    if (!groupKey) continue;

    // グループ初期化
    if (!groupedData[groupKey]) {
      groupedData[groupKey] = {
        meta: {
          source_sheet: INVOICE_SHEET_NAME,
          row_numbers: [],
          timestamp: new Date().toISOString()
        },
        data: {}
      };
      
      // 請求書キーのセット
      if (headers[keyIdx]) {
         groupedData[groupKey].data[headers[keyIdx]] = groupKey;
      }

      // 請求書情報（ルート項目）のセット
      mainIndices.forEach(idx => {
        if (headers[idx]) {
           groupedData[groupKey].data[headers[idx]] = row[idx];
        }
      });

      // --- 山本追加分：バリエーション情報をルート項目にセット ---
      vrtiIndices.forEach(idx => {
        if (headers[idx]) {
           groupedData[groupKey].data[headers[idx]] = row[idx];
        }
      });
      // ---------------------------------------------------

      // 配列の初期化
      groupedData[groupKey].data[KEY_NAME_DETAILS] = [];
      groupedData[groupKey].data[KEY_NAME_URLS] = [];
    }

    // 行番号の記録
    groupedData[groupKey].meta.row_numbers.push(currentRowNum);

    // 明細情報の追加（オブジェクト作成）
    const detailObj = {};
    let hasDetail = false; // 明細データがあるかチェック用
    dtlIndices.forEach(idx => {
      if (headers[idx]) {
        detailObj[headers[idx]] = row[idx];
        if (row[idx] !== "") hasDetail = true; // 何かしら値があれば有効とする
      }
    });
    // 明細情報が完全に空でなければ追加（要件に応じて調整可）
    groupedData[groupKey].data[KEY_NAME_DETAILS].push(detailObj);

    // URLの追加
    if (row[urlIdx]) {
      // URLの重複を防ぎたい場合は includes チェックを入れる
      if (!groupedData[groupKey].data[KEY_NAME_URLS].includes(row[urlIdx])) {
        groupedData[groupKey].data[KEY_NAME_URLS].push(row[urlIdx]);
      }
    }
  }

  // 5. 送信処理
  const allGroups = Object.values(groupedData);
  const getJstTime = () => new Date().toLocaleTimeString('en-US', {
      timeZone: 'Asia/Tokyo', hour12: false, hour: '2-digit', minute: '2-digit', second: '2-digit', fractionalSecondDigits: 3
  });

  console.log(`合計 ${allGroups.length} 件の請求書データを送信します。`);

  for (const group of allGroups) {
    group.data['処理開始'] = getJstTime();

    const options = {
      'method': 'post',
      'contentType': 'application/json',
      'payload': JSON.stringify(group)
    };

    try {
      const response = UrlFetchApp.fetch(WEBHOOK_URL, options);
      // キー情報を使ってログ出力
      const keyName = headers[keyIdx] || "Key";
      console.log(`Key: ${group.data[keyName]} (行数: ${group.meta.row_numbers.length}) 送信成功`);
    } catch (e) {
      console.error(`送信エラー: ${e.toString()}`);
    }
  }
  
  console.log("--- 完了 ---");

}

  const values = sheetData.getRange(startRow, 1, numRows, lastCol).getValues();

  // 4. グルーピング処理
  const groupedData = {};

  for (let i = 0; i < values.length; i++) {
    const row = values[i];
    const currentRowNum = startRow + i;
    
    // キー取得
    const groupKey = row[keyIdx];
    
    // キーが空の場合はスキップ
    if (!groupKey) continue;

    // グループ初期化
    if (!groupedData[groupKey]) {
      groupedData[groupKey] = {
        meta: {
          source_sheet: INVOICE_SHEET_NAME,
          row_numbers: [],
          timestamp: new Date().toISOString()
        },
        data: {}
      };
      
      // 請求書キーのセット
      if (headers[keyIdx]) {
         groupedData[groupKey].data[headers[keyIdx]] = groupKey;
      }

      // 請求書情報（ルート項目）のセット
      mainIndices.forEach(idx => {
        if (headers[idx]) {
           groupedData[groupKey].data[headers[idx]] = row[idx];
        }
      });

      // 配列の初期化
      groupedData[groupKey].data[KEY_NAME_DETAILS] = [];
      groupedData[groupKey].data[KEY_NAME_URLS] = [];
    }

    // 行番号の記録
    groupedData[groupKey].meta.row_numbers.push(currentRowNum);

    // 明細情報の追加（オブジェクト作成）
    const detailObj = {};
    let hasDetail = false; // 明細データがあるかチェック用
    dtlIndices.forEach(idx => {
      if (headers[idx]) {
        detailObj[headers[idx]] = row[idx];
        if (row[idx] !== "") hasDetail = true; // 何かしら値があれば有効とする
      }
    });
    // 明細情報が完全に空でなければ追加（要件に応じて調整可）
    groupedData[groupKey].data[KEY_NAME_DETAILS].push(detailObj);

    // URLの追加
    if (row[urlIdx]) {
      // URLの重複を防ぎたい場合は includes チェックを入れる
      if (!groupedData[groupKey].data[KEY_NAME_URLS].includes(row[urlIdx])) {
        groupedData[groupKey].data[KEY_NAME_URLS].push(row[urlIdx]);
      }
    }
  }

  // 5. 送信処理
  const allGroups = Object.values(groupedData);
  const getJstTime = () => new Date().toLocaleTimeString('en-US', {
      timeZone: 'Asia/Tokyo', hour12: false, hour: '2-digit', minute: '2-digit', second: '2-digit', fractionalSecondDigits: 3
  });

  console.log(`合計 ${allGroups.length} 件の請求書データを送信します。`);

  for (const group of allGroups) {
    group.data['処理開始'] = getJstTime();

    const options = {
      'method': 'post',
      'contentType': 'application/json',
      'payload': JSON.stringify(group)
    };

    try {
      const response = UrlFetchApp.fetch(WEBHOOK_URL, options);
      // キー情報を使ってログ出力
      const keyName = headers[keyIdx] || "Key";
      console.log(`Key: ${group.data[keyName]} (行数: ${group.meta.row_numbers.length}) 送信成功`);
    } catch (e) {
      console.error(`送信エラー: ${e.toString()}`);
    }
  }
  
  console.log("--- 完了 ---");
}

function doPost(e) {
  try {
    const jsonBody = JSON.parse(e.postData.contents);
    const status = jsonBody.status;
    const params = jsonBody.value; 
    const message = jsonBody.message; 

    if (!params) {
      throw new Error("JSON内に 'value' データが見つかりません。");
    }

    if (status === 'Success') {
      onSuccess(params);
    } 
    else if (status === 'Error') {
      console.warn("APIからエラーが返されました。メッセージ: " + message);
      console.log("受け取ったデータ(params): " + JSON.stringify(params));
    } 
    else {
      throw new Error(`成功でもエラーでもないステータス： ${status}`);
    }

    // ★★★ ここを追加！ n8nに「OK」と返します ★★★
    return ContentService.createTextOutput(JSON.stringify({ result: "success" }))
                         .setMimeType(ContentService.MimeType.JSON);

    // ★★★ エラー時も一応レスポンスを返しておくと親切です ★★★

  }

 */
function updateInputSheetToCyakusyu(ss, params) {
  const inputSheet = ss.getSheetByName("入力");
  if (!inputSheet) return;

  // 1. ヘッダー行（6行目）を読み込んで、各列の場所を特定する
  const headerRow = 6;
  const lastCol = inputSheet.getLastColumn();
  const headers = inputSheet.getRange(headerRow, 1, 1, lastCol).getValues()[0];

  // 各列のインデックス（0番目から数えた番号）を探す
  const keyColIdx = headers.indexOf("管理表CD+予実No+提出日時");
  const dxColIdx = headers.indexOf("着手済フラグ");

  // 見つからなかった場合の安全策
  if (keyColIdx === -1 || dxColIdx === -1) {
    console.error("必要な列（管理表CD+予実No+提出日時 または 着手済フラグ）が見つかりません。");
    return;
  }

  // 2. n8nから届いたキーを整形
  let targetKey = params['請求書キー']; 
  if (!targetKey) return;
  targetKey = String(targetKey).trim();

  const lastRow = inputSheet.getLastRow();
  if (lastRow < 9) return;

  // 3. 判定に必要な列（A列〜最大列まで）のデータを一括取得
  // ※1列ずつ取得するより、行全体のデータを一度に取るほうが処理が速いです
  const dataRange = inputSheet.getRange(9, 1, lastRow - 8, lastCol);
  const values = dataRange.getValues();

  let updatedCount = 0;
  for (let i = 0; i < values.length; i++) {
    let sheetKey = String(values[i][keyColIdx]).trim();

    if (sheetKey === targetKey) {
      // DX列（128列目）に「着手」を書き込む
      inputSheet.getRange(i + 9, dxColIdx + 1).setValue("着手");
      updatedCount++;
    }
  }

  if (updatedCount > 0) {
    console.log(`成功: キー '${targetKey}' の行を ${updatedCount} 件更新しました。`);
  } else {
    console.warn(`一致なし: キー自体がありません。`);
  }
}


/**
 * 新しい行を最終行の下に追加するロジック
 */
function appendSheetRow(sheet, headers, params) {
  // ヘッダーの並び順に基づいて、書き込むデータの配列を作成する
  const newRowData = headers.map((headerName) => {
    
    // JSON(params)の中に、シートのヘッダー名と同じキーがあるか確認
    if (params.hasOwnProperty(headerName)) {
      let newVal = params[headerName];
      
      // 数式（=）で始まる文字列から = を除去（既存ロジックを継承）
      if (typeof newVal === 'string' && newVal.startsWith('=')) {
        newVal = newVal.substring(1);
      }
      return newVal;
    } else {
      // n8nからデータが届いていない列は空欄にする
      return "";
    }
  });

  // 2. B列（4列目）において、値が入っている一番下の行番号を取得する
  const lastRowB = sheet.getRange(sheet.getMaxRows(), 2).getNextDataCell(SpreadsheetApp.Direction.UP).getRow();
  const targetRow = lastRowB + 1; // 書き込みたいのはその次の行

  // 3. 指定した行にデータを流し込む
  // 1行分 × ヘッダーの数分 の範囲を指定して、配列をセットする
  sheet.getRange(targetRow, 1, 1, newRowData.length).setValues([newRowData]);  
  console.log(`Success: Appended new row to sheet '${sheet.getName()}'`);
}

function getConcurTokenFromSheet() {
  // 1. Access the Spreadsheet and the specific Sheet
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("認証情報"); // Make sure this matches your tab name exactly

  if (!sheet) {
    console.error("Sheet 'Config' not found!");
    return;
  }

  // 2. Read the credentials from Column B
  // assuming B1=ID, B2=Secret, B3=User, B4=Pass
  var clientId = sheet.getRange("C3").getValue();
  var clientSecret = sheet.getRange("C4").getValue();
  var username = sheet.getRange("C5").getValue();
  var password = sheet.getRange("C6").getValue();

  // 3. Define the Concur Token Endpoint
  var tokenUrl = 'https://us-impl.api.concursolutions.com/oauth2/v0/token';

  // 4. Prepare the payload
  var payload = {
    'client_id': clientId,
    'client_secret': clientSecret,
    'grant_type': 'password',
    'username': username,
    'password': password
  };

  // 5. Make the API Call
  var options = {
    'method': 'post',
    'payload': payload,
    'muteHttpExceptions': true
  };

  var response = UrlFetchApp.fetch(tokenUrl, options);
  var json = JSON.parse(response.getContentText());

  // 6. Log the result or save it
  if (response.getResponseCode() == 200) {
    console.log("Success! Token received.");
    
    // Example: Write the new token back to the sheet in Cell B6
    sheet.getRange("B6").setValue(json.access_token);
    
    return json.access_token;
  } else {
    console.error("Error:", json);
    return null;
  }
}

/**

 */
function cleanValue(val) {
  if (!val) return "";
  return String(val)
    .replace(/[！-～]/g, s => String.fromCharCode(s.charCodeAt(0) - 0xFEE0)) // 全角英数を半角へ
    .replace(/\s+/g, "") // スペース（全角・半角・改行）をすべて削除
    .trim();
}

  const masterMaps = getMasterMaps();
  const mappingRules = getMappingRules(ss);
  
  const srcValues = srcSheet.getRange(
    CONFIG.SOURCE.START_ROW, 
    1, 
    srcSheet.getLastRow() - CONFIG.SOURCE.START_ROW + 1, 
    srcSheet.getLastColumn()
  ).getValues();

  const groupedData = {};

  srcValues.forEach((srcRow, rowIdx) => {
    const targetStatus = String(srcRow[targetColIdx]).trim(); 
    if (targetStatus !== "処理対象") return; 

    const rowObj = {};
    const urlList = [];
    let groupKey = "";
    let rowType = ""; // ここに「1」や「2」が入る

    mappingRules.forEach((m) => {
      if (!m || !m.fieldName) return;

      let val = (m.rule.type === 'STATIC') ? m.rule.value : srcRow[m.rule.colIndex];
      if (val === "" || val === null || val === undefined) val = "";

      // --- 判定用フラグ(RowType)の取得と数値化 ---
      if (m.fieldName === 'RowType') {
        // 全角数字を半角にし、数字以外を除去して数値化
        rowType = String(val).replace(/[０-９]/g, s => String.fromCharCode(s.charCodeAt(0) - 0xFEE0)).replace(/\D/g, "");
      }

      if (m.fieldName === 'Custom2') {
        const colJValue = String(srcRow[9]).trim(); 
        val = ["SA", "SG", "SI", "SK"].includes(colJValue) ? "070-3329-9798" : "090-9848-2794";
      }
      if (m.fieldName === 'Custom13') {
        const cleanedVal = cleanValue(val);
        if (masterMaps.custom13.hasOwnProperty(cleanedVal)) val = masterMaps.custom13[cleanedVal];
      }
      if (m.fieldName === 'Custom14' || m.fieldName === 'LineItems.Custom15') {
        const originalVal = String(val).trim();
        val = (originalVal === "Yes") ? "Y" : "N";
      }
      if (m.fieldName === 'IsEmergencyCheckRun'){
        const originalVal = String(val).trim();
        val = (originalVal !== "") ? "Y" : "N";
      }
      if (m.fieldName === 'OrgUnit4') {
        const originalVal = String(val).trim();
        val = masterMaps.orgUnit4[originalVal] || originalVal;
      }
      if (m.fieldName.includes('ExpenseTypeCode')) {
        const originalVal = String(val).trim();
        val = expenseTypeMap[originalVal] || originalVal;
      }
      if (m.fieldName.includes('Description')) {
        val = String(val).replace(/\s+/g, "");
        if (val.length >= 37) val = val.substring(0, 36);
      }
      if (m.fieldName.includes('UnitPrice') || m.fieldName.includes('Quantity')) {
        let num = Number(val.toString().replace(/,/g, ''));
        val = isNaN(num) ? 0 : num;
      } else {
        val = formatDate(val);
      }

    // --- 原価(Allocations)に関する一括制御 ---
    const isAllocationField = m.fieldName.includes('LineItems.Allocations.');
    const gcValue = String(srcRow[columnLetterToIndex("GC")]).trim();

    if (isAllocationField) {
      // GC列が「原価」ではない場合は、このフィールドの処理自体をスキップする
      if (gcValue !== "原価") {
        return; // mappingRules.forEach の現在のループをスキップ（＝setNestedValueが呼ばれない）
      }

      // 以下、GC列が「原価」の場合のみ実行されるロジック
      const isPercentageField = m.fieldName.includes('Percentage');

      // Custom1 の処理
      if (m.fieldName === 'LineItems.Allocations.Custom1') {
        const currentCustom2 = getNestedValue(rowObj, 'LineItems.Allocations.Custom2');
        val = (currentCustom2 && String(currentCustom2).trim() !== "") ? "0216" : "";
      }

      // Custom12 の処理
      if (m.fieldName === 'LineItems.Allocations.Custom12') {
        const currentCustom11 = getNestedValue(rowObj, 'LineItems.Allocations.Custom11');
        val = (currentCustom11 && String(currentCustom11).trim() !== "") ? "0216" : "";
      }

      // 固定値項目
      if (m.fieldName === 'LineItems.Allocations.Custom13') val = "T001425";
      if (m.fieldName === 'LineItems.Allocations.Custom3') val = "S";
      
      // トリム処理
      if (m.fieldName.includes('Custom4') || m.fieldName.includes('Custom11') || isPercentageField) {
        val = String(val).trim();
      }
    }
      if (m.fieldName === '請求書キー') groupKey = val;
      if (m.fieldName === 'URL') {
        if (val) urlList.push(val);
      } else {
        setNestedValue(rowObj, m.fieldName, val);
      }
    });

    if (!groupKey) return;

    if (!groupedData[groupKey]) {
      groupedData[groupKey] = {
        meta: { source_sheet: CONFIG.SOURCE.SHEET_NAME, row_numbers: [], URL: [] },
        data: {} 
      };
    }

    const group = groupedData[groupKey];
    
    // --- 数値（1＝親、2以上＝子）による上書き制御 ---
    Object.keys(rowObj).forEach(key => {
      const rowTypeNum = Number(rowType);

      if (key === 'LineItems') {
        // 明細は「親・子」関係なくすべて配列に追加
        if (!group.data.LineItems) group.data.LineItems = [];
        let lineItem = rowObj.LineItems;
        if (lineItem.Allocations && !Array.isArray(lineItem.Allocations)) {
          lineItem.Allocations = [lineItem.Allocations];
        }
        group.data.LineItems.push(lineItem);
        
      } else if (key !== 'RowType') {
        // ヘッダー情報の処理
        const isParent = (rowTypeNum === 1);
        const isEmptyHeader = (!group.data[key]);

        // 「親(1)の行」のデータ、または「まだヘッダーが空」の場合のみ採用
        if (isParent || isEmptyHeader) {
          if (rowObj[key] !== undefined && rowObj[key] !== null) {
            group.data[key] = rowObj[key];
          }
        }
      }
    });

    urlList.forEach(url => {
      if (!group.meta.URL.includes(url)) group.meta.URL.push(url);
    });
    group.meta.row_numbers.push(CONFIG.SOURCE.START_ROW + rowIdx);
    });

    const creds = getCredentials(ss); 

    // ペイロードの組み立て
    const finalPayloads = Object.values(groupedData).map(group => {
    // 処理開始時間を追加
    group.data['処理開始'] = getJstTime(); 
    group.data['URL'] = group.meta.URL; 

    return {
      data: group.data,
      meta: {
        source_sheet: group.meta.source_sheet,
        row_numbers: group.meta.row_numbers,
        concur_id: creds.concur_id,
        katsubun_pw: creds.katsubun_pw
      }
    };
  });

  executePost(finalPayloads);
}


/**
 * n8nへPOST送信
 */
function executePost(payloadList) {
  if (payloadList.length === 0) {
    console.log("送信対象データがありませんでした。");
    return;
  }

  payloadList.forEach(payload => {
    // 送信データを文字列化
    const body = JSON.stringify(payload);
    
    const options = {
      method: 'post',
      contentType: 'application/json',
      payload: body,                   // 文字列化したデータを渡す
      muteHttpExceptions: true
    };
    
    try {
      const response = UrlFetchApp.fetch(CONFIG.WEBHOOK_URL, options);
      const code = response.getResponseCode();
      const resText = response.getContentText();
      
      if (code === 200) {
        console.log(`送信成功: ${body.substring(0, 100)}...`);
      } else {
        // n8nからのエラーメッセージをログに出す
        console.error(`n8nエラー (Code: ${code}): ${resText}`);
      }
    } catch (e) {
      console.error(`通信失敗: ${e.message}`);
    }
  });
  
  Browser.msgBox(`${payloadList.length}件の送信処理が終わりました。n8n側の結果を確認してください。`);
}


/**
 * ドット区切りパスに値をセットする（ネスト構造作成）
 */
function setNestedValue(obj, path, value) {
  const keys = path.split('.');
  let current = obj;
  while (keys.length > 1) {
    const key = keys.shift();
    if (!current[key]) current[key] = {};
    current = current[key];
  }
  current[keys.shift()] = value;
}

/**
 * マッピングシートからルールを取得
 */
function getMappingRules(ss) {
  const mapSheet = ss.getSheetByName(CONFIG.MAPPING.SHEET_NAME);
  if (!mapSheet) return [];
  
  const mapData = mapSheet.getRange(
    CONFIG.MAPPING.START_ROW, 
    1, 
    mapSheet.getLastRow() - CONFIG.MAPPING.START_ROW + 1, 
    4
  ).getValues();
  
  const rules = [];
  mapData.forEach(row => {
    const match = row[CONFIG.MAPPING.COL_NO].toString().match(/#(\d+)/);
    if (match) {
      const index = parseInt(match[1], 10) - 1;
      rules[index] = {
        fieldName: row[CONFIG.MAPPING.COL_FIELD],
        rule: parseRule(row[CONFIG.MAPPING.COL_RULE])
      };
    }
  });
  return rules;
}

/**
 * ルール文字列を解析
 */
function parseRule(ruleStr) {
  if (!ruleStr) return { type: 'EMPTY' };
  const str = String(ruleStr);
  const match = str.match(/^\{\{([A-Za-z]+)\}\}$/);
  if (match) {
    return { type: 'COLUMN_REF', colIndex: columnLetterToIndex(match[1].toUpperCase()) };
  }
  return { type: 'STATIC', value: str };
}

/**
 * 列記号をインデックスに変換
 */
function columnLetterToIndex(letter) {
  let column = 0;
  for (let i = 0; i < letter.length; i++) {
    column += (letter.charCodeAt(i) - 64) * Math.pow(26, letter.length - i - 1);
  }
  return column - 1;
}

/**
 * 日付形式の変換
 */
function formatDate(dateVal) {
  if (!dateVal) return "";
  let strVal = dateVal.toString().trim();
  if (/^\d{1,4}\/\d{1,2}\/\d{1,2}$/.test(strVal)) {
    return strVal.replace(/\//g, '-');
  }
  if (Object.prototype.toString.call(dateVal) === "[object Date]") {
    return Utilities.formatDate(dateVal, Session.getScriptTimeZone(), "yyyy-MM-dd");
  }
  return dateVal;
}

/**
 * ドット区切りパスから値を取得する（ネスト構造読み取り用）
 */
function getNestedValue(obj, path) {
  const keys = path.split('.');
  let current = obj;
  for (const key of keys) {
    if (current[key] === undefined) return undefined;
    current = current[key];
  }
  return current;
}

/**
 * パラメータシートから認証情報を取得
 */
function getCredentials(ss) {
  const mapSheet = ss.getSheetByName(CONFIG.MAPPING.SHEET_NAME);
  // シートがない場合、両方空で返す
  if (!mapSheet) return { concur_id: "", katsubun_pw: "" };

  // 2行目の1列目(A)から2列分(A, B)を取得
  const creds = mapSheet.getRange(2, 1, 1, 2).getValues()[0];

  return {
    concur_id: String(creds[0] || "").trim(),    // A2
    katsubun_pw: String(creds[1] || "").trim()   // B2
  };
}
